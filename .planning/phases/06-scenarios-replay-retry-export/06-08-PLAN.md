    ```

      (s) => s.machineState,
      (cur, prev) => {
        if (cur === 'w-cyklu' && prev !== 'w-cyklu') {
          clearTimeout(_cycleEndHandle);
          _cycleEndHandle = setTimeout(() => {
            // Defensive: tylko jeśli wciąż w-cyklu (nie zatrzymane przez awarię)
            if (this.store.getState().machineState === 'w-cyklu') {

    ));
    ```
