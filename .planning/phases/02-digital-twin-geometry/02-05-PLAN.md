**Implementacja per RESEARCH §"Pivot patterns" + D-Phase2-06 + UI-SPEC §Default Pose:**

```js
/**
 * TWIN-05: Osłona przednia ruchoma — pivot u GÓRY (zawias). Otwiera się do góry (rotacja wokół X).
 * Pre-translate geometry (RESEARCH §Pivot pre-translate gotcha): origin grupy = górna krawędź osłony.
 *
 * Default pose: closed (rot.x=0). Phase 3+ wykona gsap.to(group.rotation, {x: -Math.PI/2}) na "open".
 */
_buildFrontGuard() {
  const guardGroup = new THREE.Group();
  guardGroup.position.set(0, 5, 1.5);  // pozycja zawiasu (D-Phase2-04)
  this.group.add(guardGroup);

  const guardGeo = new THREE.BoxGeometry(2.5, 1.8, 0.05);
  // Pre-translate: origin grupy ma być GÓRNĄ krawędzią osłony.
  // Geometria domyślnie ma origin w środku (Y∈[-h/2, h/2]). Po translate(0, -h/2, 0) origin staje się górną krawędzią.
  guardGeo.translate(0, -0.9, 0);

  const guard = new THREE.Mesh(guardGeo, this.matGuardOrange);
  guard.castShadow = true;
  guard.receiveShadow = true;
  guardGroup.add(guard);

  // Rejestrujemy GRUPĘ (nie sam mesh) jako interactable — userData siedzi na grupie, mesh.material clone'a baseMaterial.
  // Phase 3 RaycastController z `recursive: true` trafi w `guard` (mesh dziecko) i poprzez parent traversal odczyta userData z grupy.
  // ALTERNATYWA: rejestrujemy mesh, userData na meshu — wtedy poses dotyczą rotacji mesh'a, ale mesh siedzi w grupie...
  //
  // Decyzja: rejestruję GRUPĘ jako interactable mesh (Three.js Group dziedziczy po Object3D, ma userData/material slots).
  //          Phase 3 raycaster trafi w `guard` mesh (dziecko grupy); wymaga walk-up do parent gdzie jest userData.
  //
  // Aby UPROŚCIĆ: rejestruję `guard` MESH jako interactable (raycaster trafia bezpośrednio), a userData.poses
  // odnosi się do rotacji RODZICA (`guardGroup`). Tu jest niejednoznaczność — Phase 3 wykona
  // `guardGroup = guard.parent; gsap.to(guardGroup.rotation, ...)`.
  //
  // Final decision: rejestruję `guard` MESH; userData.poses jest semantyką "rotacja rodzica grupy".
  // Komentarz objaśniający dla Phase 3:
  //   "TWIN-05/02/09: poses dotyczą rotacji mesh.parent (pivot-group). Phase 3 animator: const pivot = mesh.parent; gsap.to(pivot.rotation, poses[targetPose].rot)."

  this._registerInteractable({
    mesh: guard,
    id: 'oslona-przednia',
    kind: 'manipulation',
    baseMaterial: this.matGuardOrange,
    poses: {
      closed: { rot: { x: 0, y: 0, z: 0 } },
      open:   { rot: { x: -Math.PI / 2, y: 0, z: 0 } },
    },
  });
  // Default pose: closed → guardGroup.rotation.x = 0 (już domyślne).
  // restPosition zarejestrowany przez _registerInteractable z `guard.position` (po pre-translate: (0, -0.9, 0)).
  // To NIE jest restPosition pivota — to restPosition mesha w lokalnym frame'a pivota.
  // OK — restPosition w plain object format pozostaje przyjętym kontraktem (CRIT-7).
}
```

**Uwaga implementacyjna**: powyższy komentarz `Decyzja:` musi pojawić się w kodzie inline (JSDoc nad metodą). To wyjaśnia konwencję dla Phase 3 — rejestrowany mesh `guard`, ale `userData.poses` odnosi się do rotacji `mesh.parent` (grupy pivota).

  <action>

**Implementacja per RESEARCH §"ExtrudeGeometry — wyłącznik główny" + §"LatheGeometry":**

```js
/**
 * TWIN-09: Wyłącznik główny — cylindryczny korpus (LatheGeometry) + pokrętło z karbami
 * (ExtrudeGeometry z 4 holes Path).
 *
 * Pivot pokrętła: oś Z lokalnego frame'a (ExtrudeGeometry ekstruduje wzdłuż +Z, więc origin Shape'a (0,0)
 * jest osią obrotu pokrętła).
 *
 * Default pose: off (rot.z=0). Phase 3+ tween do rot.z=Math.PI/2 na "on".
 *
 * Pozycja per D-Phase2-04: świat (2.5, 4, -0.5) — bok korpusu prawy.
 */
_buildMainSwitch() {
  const switchGroup = new THREE.Group();
  switchGroup.position.set(2.5, 4, -0.5);
  this.group.add(switchGroup);

  // 1. Korpus wyłącznika (LatheGeometry, decorative) — krótki cylindryczny "kubek" do montażu pokrętła
  const bodyPoints = [
    new THREE.Vector2(0.0,  0.0),
    new THREE.Vector2(0.18, 0.0),
    new THREE.Vector2(0.18, 0.10),
    new THREE.Vector2(0.0,  0.10),
  ];
  const bodyGeo = new THREE.LatheGeometry(bodyPoints, 24);
  // Lathe rotuje wokół Y → korpus stoi na Y=0..0.10.
  // Aby korpus był wbudowany w ścianę boczną prasy (płaszczyzna XZ panel) — rotujemy całość.
  // Bok prasy = płaszczyzna +X (świat); chcemy korpus wystający z +X kierunku.
  // Default lathe ma oś Y; po rotateZ(Math.PI/2) oś staje się X. To matchuje wystającemu z boku.
  bodyGeo.rotateZ(-Math.PI / 2);  // korpus wystaje wzdłuż +X
  const body = new THREE.Mesh(bodyGeo, this.matSwitchBody);
  body.castShadow = true;
  switchGroup.add(body);
  // Korpus NIE jest interactable (decorative, shared this.matSwitchBody).

  // 2. Pokrętło — ExtrudeGeometry koło z 4 prostokątnymi karbami (RESEARCH recipe)
  const knobShape = new THREE.Shape();
  knobShape.absellipse(0, 0, 0.15, 0.15, 0, Math.PI * 2, false);  // outer circle r=0.15

  // 4 karby co 90° jako Path holes (rowki w pokrętle, RESEARCH recipe)
  for (let i = 0; i < 4; i++) {
    const angle = (Math.PI / 2) * i;
    const cx = Math.cos(angle) * 0.10;
    const cy = Math.sin(angle) * 0.10;
    const notch = new THREE.Path();
    notch.moveTo(cx - 0.015, cy - 0.025);
    notch.lineTo(cx + 0.015, cy - 0.025);
    notch.lineTo(cx + 0.015, cy + 0.025);
    notch.lineTo(cx - 0.015, cy + 0.025);
    notch.lineTo(cx - 0.015, cy - 0.025);
    knobShape.holes.push(notch);
  }

  const knobGeo = new THREE.ExtrudeGeometry(knobShape, {
    depth: 0.04,
    bevelEnabled: true,
    bevelThickness: 0.003,
    bevelSize: 0.003,
    bevelSegments: 2,
    curveSegments: 24,
  });
  // ExtrudeGeometry domyślnie ekstruduje wzdłuż +Z. Podobnie do korpusu — rotujemy żeby pokrętło
  // wystawało wzdłuż +X (na zewnątrz prasy).
  knobGeo.rotateY(Math.PI / 2);
  // Po rotateY oś Z lokalnego pokrętła → oś X świata (po przeniesieniu przez switchGroup).
  // Pivot pokrętła = oś Z lokalnego frame'a Shape'a (0,0) → po rotateY staje się X świata.
  // D-Phase2-06: poses { off: {rot:{z:0}}, on: {rot:{z:Math.PI/2}} } — UWAGA: poses spec używa Z, ale
  // physical axis po rotateY to X. **Zaczynamy z poses używającymi axis lokalnego mesha (PRZED rotateY).**
  // Phase 3 animator: gsap.to(knob.rotation, {z: Math.PI/2}) — rotuje pokrętło wokół osi prostopadłej do bodyOf.

  const knob = new THREE.Mesh(knobGeo, this.matSwitchBody);
  knob.position.set(0.10, 0, 0);  // pokrętło wystaje z korpusu w +X
  knob.castShadow = true;
  switchGroup.add(knob);

  this._registerInteractable({
    mesh: knob,
    id: 'wylacznik-glowny',
    kind: 'manipulation',
    baseMaterial: this.matSwitchBody,
    poses: {
      off: { rot: { x: 0, y: 0, z: 0 } },
      on:  { rot: { x: 0, y: 0, z: Math.PI / 2 } },
    },
  });
  // Default pose: off → knob.rotation.z = 0 (default).
  // **Reading note dla Phase 3 animator**: poses.{off|on}.rot.z odnosi się do `knob.rotation.z`
  // (rotacja sama mesh, nie parent grupy — różnica vs oslona-przednia).
  // Pivot pokrętła to origin Shape'a (0,0) → centerline pokrętła = oś rotacji.
}
```

**Komentarz nad metodą** musi explicit wyjaśnić różnicę: `oslona-przednia.poses` → rotuje `mesh.parent` (grupa pivota); `wylacznik-glowny.poses` → rotuje `mesh` sam (knob centerline jako pivot Shape'a). /**
```js
 * TWIN-02: Dźwignia sprzęgła ruchoma — pivot u PODSTAWY przy wale.
 * Pre-translate geometry: origin grupy = dolny koniec dźwigni (punkt obrotu).
 * Pręt (CylinderGeometry) + gałka (SphereGeometry) na końcu — visual cue gdzie operator chwyta (CONTEXT discretion).
 *
 * Default pose: released (rot.z=0, dźwignia pionowa). Phase 3+ tween do rot.z=0.7 na "engaged".
 *
 * Pozycja per D-Phase2-04: świat (-3, 7, 0.5).
 *
 * Konwencja poses: rotacja MESH'A (pręt), bo operator widzi rotację samego pręta z gałką wokół podstawy.
 * To różni się od TWIN-05 (rotuje parent) — udokumentowane w 02-06 SUMMARY.
 *
 * ALTERNATYWA: rotacja parent grupy (jak TWIN-05). Wybieram konsystencję ze TWIN-05 (parent rotation):
 * leverGroup jest pivotem; pręt + gałka siedzą w grupie z pre-translated geometry.
 */
_buildClutchLever() {
  const leverGroup = new THREE.Group();
  leverGroup.position.set(-3, 7, 0.5);  // podstawa dźwigni przy wale
  this.group.add(leverGroup);

  // Pręt — pre-translate: origin grupy = dolny koniec
  const leverGeo = new THREE.CylinderGeometry(0.05, 0.05, 1.5, 12);
  leverGeo.translate(0, 0.75, 0);  // origin po translate: dolny koniec dźwigni
  const lever = new THREE.Mesh(leverGeo, this.matBrakeSteel);  // używamy palety stali (pręt)
  lever.castShadow = true;
  leverGroup.add(lever);

  // Gałka — wizualna podpowiedź gdzie chwytać (CONTEXT discretion rekomendacja)
  const knobGeo = new THREE.SphereGeometry(0.1, 16, 12);
  const knob = new THREE.Mesh(knobGeo, this.matSafetyButtonGreen);  // zielona gałka — accent czytelności
  knob.position.set(0, 1.5, 0);  // koniec dźwigni w lokalnym frame'a grupy
  knob.castShadow = true;
  leverGroup.add(knob);
  // Gałka NIE jest osobnym interactable — dzieli interactable z prętem (cały pręt + gałka = 'dzwignia-sprzegla').

  // PRIMARY mesh dla interactable = `lever` (pręt). Phase 3 raycaster trafi w pręt LUB gałkę;
  // raycast walk-up do leverGroup, identyfikacja interactable przez userData ID.
  this._registerInteractable({
    mesh: lever,
    id: 'dzwignia-sprzegla',
    kind: 'manipulation',
    baseMaterial: this.matBrakeSteel,
    poses: {
      released: { rot: { x: 0, y: 0, z: 0 } },
      engaged:  { rot: { x: 0, y: 0, z: 0.7 } },
    },
  });
  // poses semantyka per UI-SPEC §Default Pose: rotacja LEVERGROUP (pivot u podstawy) wokół Z.
  // Phase 3 animator: const pivot = lever.parent; gsap.to(pivot.rotation, {z: 0.7}).
  // Komentarz w UI-SPEC i CONTEXT D-Phase2-06 to potwierdzają.
}
```
